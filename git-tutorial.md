
```bash
# コミットの修正
git commit --amend --no-edit

# コミットのリセット
git reset

# 「コミットの変更を打ち消す」コミット
git revert

# コミットの実行
git commit

# ブランチの作成
git branch

# ブランチの切り替え
git checkout

# ブランチの統合
git merge

# リモートレポジトリの設定
git remote

# リモートレポジトリの同期
git pull

# 変更をアップロードする
git push

# プルリクエスト：変更依頼
git request-pull

# レポジトリをコピー
git clone

# gitの初期化・設定開始
git init

# ステージングエリアに追加
git add

# ファイルとindex情報の削除
git rm

# ワークツリーのステータスを表示
git status

# ログを表示
git log

# 設定周りの確認・変更
git config

# ファイルの差分を表示
git diff

# fast forward only. 変更のない統合先ブランチにマージ（参考）
--ff-only

# コミットメッセージの1行のみの一覧表示
--oneline
